== Variables

[source,yaml]
.httpd/vars/main.yml
----
httpd_version: 2.3
----

[source,yaml]
.httpd/tasks/main.yml
----
- name: Ensure httpd is installed
  yum: name= "httpd"
            state= present
            version= {{ httpd_version }}
----
